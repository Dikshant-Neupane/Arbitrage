# ============================================

# --------------------------------------------
# Core Web Scraping Libraries
# --------------------------------------------

beautifulsoup4==4.14.3
# Beautiful Soup is a library for pulling data out of HTML and XML files.
# Used for: Parsing real estate website HTML and extracting property data
# Docs: https://www.crummy.com/software/BeautifulSoup/bs4/doc/

requests==2.32.5
# Requests allows you to send HTTP requests easily.
# Used for: Fetching web pages from real estate websites
# Docs: https://requests.readthedocs.io/

soupsieve==2.8.2
# CSS selector library for Beautiful Soup
# Used for: Advanced HTML element selection using CSS selectors
# Auto-installed with beautifulsoup4

# --------------------------------------------
# Data Analysis & Manipulation
# --------------------------------------------

pandas==2.3.3
# Pandas provides data structures and analysis tools for Python
# Used for: 
#   - Loading CSV files (raw and cleaned data)
#   - Data cleaning and transformation
#   - Calculating price per square foot
#   - Statistical analysis of property prices
# Docs: https://pandas.pydata.org/docs/

numpy==2.4.1
# NumPy is the fundamental package for numerical computing
# Used for: 
#   - Mathematical operations on property data
#   - Array manipulations
#   - Statistical calculations
# Docs: https://numpy.org/doc/

# --------------------------------------------
# Date & Time Handling
# --------------------------------------------

python-dateutil==2.9.0.post0
# Extensions to the standard Python datetime module
# Used for: Parsing listing dates and timestamps from websites
# Docs: https://dateutil.readthedocs.io/

pytz==2025.2
# World timezone definitions for Python
# Used for: Handling timezones in property listing dates
# Docs: https://pythonhosted.org/pytz/

tzdata==2025.3
# IANA time zone database for Python
# Used for: Timezone data support for cross-platform compatibility
# Auto-installed with pandas

# Python package for providing Mozilla's CA Bundle
# Used for: SSL certificate verification for secure HTTPS connections
